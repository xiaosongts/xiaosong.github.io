Spring中使用

在 Spring 中注册 jasypt bean 没有什么特别的，因为 jasypt 中的所有加密工具(digester 和加密器)都有足够的设计，可以从 Spring 应用程序上下文正确地实例化和依赖注入。

1、依赖：

<dependency>
    <groupId>org.jasypt</groupId>
    <artifactId>jasypt-spring4</artifactId>
    <version>1.9.3</version>
</dependency>

2、在spring配置文件中配置：

	<bean id="environmentVariablesConfiguration"
		  class="org.jasypt.encryption.pbe.config.EnvironmentStringPBEConfig">
		<property name="algorithm" value="PBEWithMD5AndDES" />
		<property name="password" value="cdf_So@2024" />
	</bean>
	<bean id="configurationEncryptor"
		  class="org.jasypt.encryption.pbe.StandardPBEStringEncryptor">
		<property name="config" ref="environmentVariablesConfiguration" />
	</bean>
	<bean id="propertyConfigurer"
		  class="org.jasypt.spring4.properties.EncryptablePropertyPlaceholderConfigurer">
		<constructor-arg ref="configurationEncryptor" />
		<property name="locations">
			<list>
				<value>classpath:jdbc.properties</value>
				<value>classpath:redis.properties</value>
				<value>classpath:config.properties</value>
				<value>classpath:xxl-job-executor.properties</value>
			</list>
		</property>
	</bean>
